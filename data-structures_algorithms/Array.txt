#include<iostream>
using namespace std;

{

int arr[n];

{

cin>>arr[i];
}

cout<<"The array elements are: [ ";

for(i=0; i<n; i++)
{
cout<<arr[i]<<" ";
}

cout<<"]"<<endl;

}
